SELECT A.HY_TIT, B.DL_TIT, C.CD_TIT, D.OTHER_TIT, A.HY_TIT+B.DL_TIT+C.CD_TIT+D.OTHER_TIT AS Profit, E.KHFY_TIT, E.KHFY_LIRUNDAXIE
FROM HY_FYXX AS A, DL_FYXX AS B, CD_FYXX AS C, OTHER_FYXX AS D, KHFY AS E
WHERE A.HY_TD=B.DL_ID 
  And C.CD_ID = D.Other_ID 
  AND B.DL_ID=C.CD_ID
  AND A.HY_TD=E.KHFY_ID
  AND A.HY_TD=:HY_TD;


select * from Fyquery
where HY_TD=:HYTD
